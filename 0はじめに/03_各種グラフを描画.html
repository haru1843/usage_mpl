plot

plotの公式サイトはこちら

import numpy as np
import matplotlib.pyplot as plt

def func(x):
  return np.sinc(x)

def main():
  fig = plt.figure()
  ax = fig.add_subplot(111)

  xlim = [-10, 10]

  x = np.linspace(*xlim, 1000)
  y = func(x)

  ax.plot(x, y)  # plot で 折れ線を描画

  ax.grid()
  ax.set_xlim(xlim)
  plt.show()

if __name__ == '__main__':  
  main()

色

colorまたはcへ適当な文字列を渡すとその色で描画できる.

色の選択は#808080のようにカラーコードで指定したり, 色を表す文字列でも指定できます.

名前がつけられている色は公式サイトを見ると, 一発でわかります.
あと, デフォルトで順につけられていく色は, C0, C1, C2 …, のようになっています.
そのためもし, デフォルト色でつけていきたいときはcolor='C0'のように設定します.

plot(x, y, color="black")
plot(x, y, c="y")
plot(x, y, color="#FF7777")
plot(x, y, c="#3355FF")

import numpy as np
import matplotlib.pyplot as plt

def func(x):

def main():

  ax.plot(x, y, c="#22DDBB")  # c にカラーコードを渡し, 色を指定

  ax.grid()
  ax.set_xlim(xlim)
  plt.show()

if __name__ == '__main__':  
  main()

線のスタイル, 太さ

linestyle, lsで線のスタイルを変更できる.
また, linewidth, lwで線の太さを変更できる.

渡す引数とスタイルの様子は公式サイトで確認できます.

import numpy as np
import matplotlib.pyplot as plt

def func(x):
  return np.sinc(x)

def main():
  fig = plt.figure()
  ax = fig.add_subplot(111)

  xlim = [-10, 10]

  x = np.linspace(*xlim, 1000)
  y = func(x)

  ax.plot(x, y, c="#22DDBB", ls="--", lw=3)
  
  ax.grid()
  ax.set_xlim(xlim)
  plt.show()

if __name__ == '__main__':  
  main()

透明度

alphaで透明度を設定できます.

0≤alpha≤10 \leq {\rm alpha} \leq 10≤alpha≤1の範囲でalphaを選択し, 0だと完全に透明, 1に近づくほど濃くなっていきます.

import numpy as np
import matplotlib.pyplot as plt

  y = func(x)

  ax.plot(x, y, c="#22DDBB", ls="--", alpha=0.5)

  ax.grid()
  ax.set_xlim(xlim)
  plt.show()

if __name__ == '__main__':  
  main()

表示順を表側/裏側に持ってくる

zorderで描画順番を指定できる.
指定値はfloatで, 数値が低ければ低いほど裏側に描画され, 高ければ高い程表側に描画される.

import numpy as np
import matplotlib.pyplot as plt
import matplotlib as mpl

def func(x):
  return np.sinc(x)

def main():
  font = {"family": "IPAexGothic"}
  mpl.rc('font', **font)

  fig = plt.figure()
  ax = fig.add_subplot(111)

  xlim = [-10, 10]

  x = np.linspace(*xlim, 1000)

  ax.plot(x, func(x + 2.5), zorder=-100, label="めっちゃ後ろ")
  ax.plot(x, func(x), zorder=0, label="めっちゃ中間")
  ax.plot(x, func(x - 2.5), zorder=100, label="めっちゃ前")

  ax.legend()
  ax.grid()
  ax.set_xlim(xlim)
  plt.show()

if __name__ == '__main__':  
  main()
