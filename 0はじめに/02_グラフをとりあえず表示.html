fig/ax

figure

pyplotから, plt.figure()によって, figureを用意します.
figureは大元の大きな台紙だと思うと良いと思います.

from matplotlib import pyplot as plt

def main():
  fig = plt.figure() # figureの用意

if __name__ == '__main__':
  main()

axes

作成した大きな台紙に, グラフを書く場所を用意します.
作成したfigureから, add_subplot()メソッドを利用して, グラフを書く場所であるaxesを用意します.

from matplotlib import pyplot as plt

def main():
  fig = plt.figure()
  ax = fig.add_subplot(111) # figureにaxを用意

if __name__ == '__main__':
  main()

add_subplot()の引数についてはまたあとで説明します.

描画したものの表示

とりあえず, 表示まではできるように成っておきましょう.

from matplotlib import pyplot as plt

def main():
  fig = plt.figure()
  ax = fig.add_subplot(111)

  plt.show() # 表示

if __name__ == '__main__':
  main()

画像保存

pngで保存

from matplotlib import pyplot as plt

def main():
  fig = plt.figure()
  ax = fig.add_subplot(111)

  plt.savefig("file_name.png") # pngで保存

if __name__ == '__main__':
  main()

svgで保存

from matplotlib import pyplot as plt

def main():
  fig = plt.figure()
  ax = fig.add_subplot(111)

  plt.savefig("file_name.svg", format="svg") # svgで保存

if __name__ == '__main__':
  main()

pdfで保存

from matplotlib import pyplot as plt
# PdfPagesのインポート
from matplotlib.backends.backend_pdf import PdfPages

def main():
  fig = plt.figure()
  ax = fig.add_subplot(111)

  # pdfファイルの初期化
  pp = PdfPages('file_name.pdf')
  # figureをセーブする
  pp.savefig(fig)
  # pdfファイルをクローズする。
  pp.close()

if __name__ == '__main__':
  main()
